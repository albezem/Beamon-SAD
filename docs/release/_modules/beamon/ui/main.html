[docs]    def append_to_log(self, msg):
        """
        Appends a given message to log and timestampes it with current datetime
        :param msg: string
        """
        if len(msg) == 0:
            return
        dt = datetime.now().strftime("%d-%m-%Y_%H-%M-%S")
        self.log.append(dt + '\t' + msg)
        self.display_message(msg)

[docs]    def open_database(self, path, visualization=None, overwrite=False):
        """
        Opens a database in path. opens a new Database if overwrite is true. if File do not exist and overwrite is false
        an exeption will be thrown
        :param path: path to database file
        :type path: string
        :param visualization: Visualization object to be referenced by database (default None)
        :type visualization: Visualization
        :param overwrite: True/False
        :type overwrite: boolean
        """
        if overwrite:
            # remove database first
            if os.path.isfile(path):
                os.remove(path)
            # open new database
            self.database = Database(ram=self.ram, path_to_save_database=path)
        else:
            # open existing database
            self.database = Database(path_to_database=path, ram=self.ram)
            # self.database.sort_node_indexes()
        self.setWindowTitle(self.database.name)



[docs]    def open_database_diag(self):
        """
        Opens a user interface to load a database file
        """
        dlg = QFileDialog()
        path = os.getcwd()
        filter = "Database (*.db)"
        title = "open db file"
        fname = QFileDialog.getOpenFileName(dlg, title, path, filter)
        # init the database
        if fname[0] == "":
            # no file was chosen, init default empty database
            return
        else:
            try:
                self.database.close()
                self.open_database(path=fname[0])
                # self.database.sort_node_indexes()
                # refresh visualization if open
                for item in self.ui.mdiArea.subWindowList():
                    if item.windowTitle() == "Visualizer":
                        self.ui.mdiArea.setActiveSubWindow(item)
                        self.ui.mdiArea.closeActiveSubWindow()
                        self.calculation.data_set = self.database
                        self.startVisualizer()
                self.append_to_log("project " + str(path) + " loaded.")
            except Exception as e:
                self.append_to_log(str(e))

[docs]    def closeEvent(self, event):
        """
        controls closing signal
        :param event: the closing event
        """
        if self.confirm_exit:
            msg = QMessageBox()
            msg.setIcon(QMessageBox.Warning)
            msg.setText("Are you sure you want to close?")
            msg.setWindowTitle("Closing app")
            msg.setStandardButtons(QMessageBox.Ok | QMessageBox.Cancel)
            retval = msg.exec_()
            if retval == QMessageBox.Cancel:
                event.ignore()
            else:
                # close all opened programmes after accept
                self.ui.mdiArea.closeAllSubWindows()
                if len(self.ui.mdiArea.subWindowList()) == 0:
                    # close database connection
                    # self.database.close()
                    event.accept()
                else:
                    event.ignore()
        else:
            event.accept()

        """
        starts the application BeamSize as a subwindow
        """

        # if singleton (call only once)
        for item in self.ui.mdiArea.subWindowList():
            # print(item.windowTitle())
            if item.windowTitle() == "BeamSize":
                self.ui.mdiArea.setActiveSubWindow(item)
                return
        qsw = BeamSizeMainForm(self)
        self.ui.mdiArea.addSubWindow(qsw)
        qsw.show()
        self.ui.toolBar.addAction(self.beamSize)

        # to maximize the window on start
        qsw.setWindowState(Qt.WindowMaximized)



[docs]    def startVisualizer(self):
        """
        starts the application for visualising models
        """

        # if singleton (call only once)
        for item in self.ui.mdiArea.subWindowList():
            if item.windowTitle() == "Visualizer":
                self.ui.mdiArea.setActiveSubWindow(item)
                return
        # create the object
        self.visualizer = VisualizerMainForm(self.ui, self)
        # make it a subwindow
        self.ui.mdiArea.addSubWindow(self.visualizer)
        self.visualizer.show()
        self.ui.toolBar.addAction(self.visualizerAction)
        # to maximize the window on start
        self.visualizer.setWindowState(Qt.WindowMaximized)
