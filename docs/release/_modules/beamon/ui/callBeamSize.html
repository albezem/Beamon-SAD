[docs]    def updateResults(self):
        """
        updates all results of normalization and transformation
        :return:
        """

        self.calcMiddlePoints()
        self.__a = self.__a0 + self.__as + self.__au
        self.ui.lineEditA.setText(str(round(self.__a, 3)))
        self.__ys = (self.__a0 * self.__y0 + self.__au * self.__yu) / self.__a
        self.ui.lineEditys.setText(str(round(self.__ys, 3)))
        self.__zs = (self.__a0 * self.__z0 + self.__au * self.__zu) / self.__a
        self.ui.lineEditzs.setText(str(round(self.__zs, 3)))
        self.updateNormalizationValues()

        if np.abs(self.__ayz) < EPS:
            self.__alphaR = 0
        elif np.abs(self.__ayy - self.__azz) < EPS:
            self.__alphaR = np.pi / 4
        else:
            self.__alphaR = (1 / 2) * np.arctan(2 * self.__ayz / (self.__ayy - self.__azz))
        self.ui.lineEditAlphaRad.setText(str(round(self.__alphaR, 3)))
        self.__alphaD = self.__alphaR * (180 / np.pi)
        self.ui.lineEditAlphaGrad.setText(str(round(self.__alphaD, 3)))
        self.__iy = self.__ayy * np.sin(self.__alphaR) ** 2 + self.__azz * \
                    (np.cos(self.__alphaR) ** 2) - self.__ayz * 2 * np.sin(self.__alphaR) * np.cos(self.__alphaR)
        self.ui.lineEditIy.setText(str(round(self.__iy, 3)))
        self.__iz = self.__ayy * (np.cos(self.__alphaR) * np.cos(self.__alphaR)) + self.__azz * (
                np.sin(self.__alphaR) * np.sin(self.__alphaR)) + self.__ayz * 2 * np.sin(self.__alphaR) * np.cos(
            self.__alphaR)
        self.ui.lineEditIz.setText(str(round(self.__iz, 3)))
        self.ui.lineEditIy_2.setText(str(round(self.__ayy, 3)))
        self.ui.lineEditIz_2.setText(str(round(self.__azz, 3)))
        self.__iyz = self.__ayz * np.cos(self.__alphaR) ** 2 - np.sin(self.__alphaR) ** 2 - np.sin(
            self.__alphaR) * np.cos(self.__alphaR) * (self.__ayy - self.__azz)
        self.ui.lineEditIyz.setText(str(round(self.__iyz, 3)))
        self.__yd = -self.__ys * np.cos(self.__alphaR) - self.__zs * np.sin(self.__alphaR)
        self.ui.lineEdityD.setText(str(round(self.__yd, 3)))
        self.__zd = -self.__zs * np.cos(self.__alphaR) + self.__ys * np.sin(self.__alphaR)
        self.ui.lineEditzD.setText(str(round(self.__zd, 3)))

        self.__it = (self.__b0 * self.__t0 ** 3 + self.__hs * self.__ts ** 3 + self.__bu * self.__tu ** 3) / 3
        self.ui.lineEditIT.setText(str(round(self.__it, 3)))
        self.__wk = (-self.__a0 * self.__y0 * self.__z0 - self.__au * self.__yu * self.__zu) / self.__a
        self.ui.lineEditOmegak.setText(str(round(self.__wk, 3)))

        if np.abs(self.__iy) < EPS:
            self.__ym1 = 0
        else:
            self.__ym1 = (((self.__azw - self.__zs * self.__wk * self.__a) * np.cos(self.__alphaR) -
                           (self.__ayw - self.__ys * self.__wk * self.__a) * np.sin(
                        self.__alphaR)) / self.__iy) + self.__yd
        self.ui.lineEdityM.setText(str(round(self.__ym1, 3)))

        if np.abs(self.__iz) < EPS:
            self.__zm1 = 0
        else:
            self.__zm1 = (-((self.__ayw - self.__ys * self.__wk * self.__a) * np.cos(self.__alphaR) + (
                    self.__azw - self.__zs * self.__wk * self.__a) * np.sin(self.__alphaR)) / self.__iz) + self.__zd
        self.ui.lineEditzM.setText(str(round(self.__zm1, 3)))

        self.__iw = self.__aww - self.__wk ** 2 * self.__a - (self.__ym1 - self.__yd) ** 2 * self.__iy - \
                    (self.__zm1 - self.__zd) ** 2 * self.__iz
        self.ui.lineEditIOmega.setText(str(round(self.__iw, 3)))
        self.__ym2 = self.__ym1 * np.cos(self.__alphaR) - self.__zm1 * np.sin(self.__alphaR) + self.__ys
        self.ui.lineEdityM2.setText(str(round(self.__ym2, 3)))
        self.__zm2 = self.__zm1 * np.cos(self.__alphaR) + self.__ym1 * np.sin(self.__alphaR) + self.__zs
        self.ui.lineEditzM2.setText(str(round(self.__zm2, 3)))


    def updateNormalizationValues(self):
        self.__ayz = self.__a0 * self.__y0 * self.__z0 + self.__au * self.__yu * self.__zu - \
                     self.__ys * self.__zs * self.__a
        self.__ayy = self.__a0 * (self.__y0 ** 2 + (self.__b0 ** 2) / 12) + self.__au * \
                     (self.__yu ** 2 + self.__bu ** 2 / 12) - self.__ys ** 2 * self.__a
        self.__azz = self.__a0 * self.__z0 ** 2 + self.__au * self.__zu ** 2 + self.__as * (self.__hs ** 2) / 12 - \
                     self.__zs ** 2 * self.__a
        self.__ayw = -self.__a0 * self.__z0 * (self.__y0 ** 2 + (self.__b0 ** 2) / 12) - self.__au * self.__zu * \
                     (self.__yu ** 2 + (self.__bu ** 2) / 12)
        self.__azw = -self.__a0 * self.__y0 * self.__z0 ** 2 - self.__au * self.__yu * self.__zu ** 2
        self.__aww = self.__a0 * self.__z0 ** 2 * (
                self.__y0 ** 2 + (self.__b0 ** 2) / 12) + self.__au * self.__zu ** 2 * \
                     (self.__yu ** 2 + (self.__bu ** 2) / 12)


[docs]    def initPlotWidget(self):
        """
        initializes the plot area settings and add the ROI elements to the plotwidget
        """

        self.ui.plotWidget.setMenuEnabled(False)

        # very important in order to prevent sheering
        self.ui.plotWidget.plotItem.getViewBox().setAspectLocked()

        text = """berlappungsmodell (Dnnwandig)"""
        self.ui.plotWidget.getPlotItem().setTitle(text)

        # dont't show axes since they are different from the calculations
        self.ui.plotWidget.hideAxis('bottom')
        self.ui.plotWidget.hideAxis('left')

        # make the drawing color of the elements black
        pen = pg.mkPen('k')

        # add the bridge (Steg)
        # bridge is unmovable and placed in the origin (0, 0) translated from the lower lef conrner
        self.bridge = pg.ROI([0, 0], [0, 0], movable=False, pen=pen)

        # self.bridge.addScaleHandle([1, 1], [0, 0])
        # add top chord (Obergurt)
        self.top_chord = pg.ROI([0, 0], [0, 0], movable=False, pen=pen)
        # self.top_chord.addScaleHandle([1, 1], [0, 0])
        # add lower chord (Untergurt)
        self.lower_chord = pg.ROI([0, 0], [0, 0], movable=False, pen=pen)
        # self.lower_chord.addScaleHandle([1, 1], [0, 0])
        self.ui.plotWidget.addItem(self.bridge, lockAspect=True)
        self.ui.plotWidget.addItem(self.top_chord, lockAspect=True)
        self.ui.plotWidget.addItem(self.lower_chord, lockAspect=True)

        # drawing the coordinate system s
        self.origin = [-self.__ys, -self.__zs]

        self.s_coordinate_system = []
        # style of the arrows with pg.mkpen settings
        pen = {'color': '339933', 'width': 3.5}

        # first arrow
        r1 = pg.ROI([self.origin[0], self.origin[1]], [10, 0], pen=pen)
        r1.setAngle(180, finish=False)
        self.s_coordinate_system.append(r1)
        r2 = pg.ROI([r1.pos()[0] - r1.size()[0], r1.pos()[1]], [3, 0], pen=pen)
        r2.rotate(25, finish=False)
        self.s_coordinate_system.append(r2)
        r3 = pg.ROI([r1.pos()[0] - r1.size()[0], r1.pos()[1]], [3, 0], pen=pen)
        r3.rotate(-25, finish=False)
        self.s_coordinate_system.append(r3)

        index1 = pg.TextItem(html='<span style="color: #339933;font-size: 10px;">Yh</span>', anchor=(1, 1))
        index1.setParentItem(r2)

        # second arrow
        r4 = pg.ROI([self.origin[0], self.origin[1]], [10, 0], pen=pen)
        r4.rotate(-90, finish=False)
        self.s_coordinate_system.append(r4)
        r5 = pg.ROI([r4.pos()[0], r4.pos()[1] - r4.size()[0]], [3, 0], pen=pen)
        r5.rotate(90 - 25, finish=False)
        self.s_coordinate_system.append(r5)
        r6 = pg.ROI([r4.pos()[0], r4.pos()[1] - r4.size()[0]], [3, 0], pen=pen)
        r6.rotate(90 + 25, finish=False)
        self.s_coordinate_system.append(r6)

        index2 = pg.TextItem(html='<span style="color: #339933;font-size: 11px;">Zh</span>', anchor=(0, 0))
        index2.setParentItem(r5)

        # add arrows to plot
        for roi in self.s_coordinate_system:
            self.ui.plotWidget.addItem(roi)

        # for interactive plot controle the signals for position changes
        # TODO make interaction works and change movable=True in ROIs
        # TODO interaction needs snipping to grid activated
        self.bridge.sigRegionChangeFinished.connect(self.updateInputsFromPlot)
        self.top_chord.sigRegionChangeFinished.connect(self.updateInputsFromPlot)
        self.lower_chord.sigRegionChangeFinished.connect(self.updateInputsFromPlot)
        self.ui.plotWidget.sigRangeChanged.connect(self.updateCoordinateSystems)

        self.updateGraph()

        """
        updates the cross section origins position and orientation according to (ys, zs) and alpha respectively
        also uses the view range as scale for the size of the Arrows of the coordinate system
        :return:
        """

        # update origin
        self.origin = [-self.__ys, -self.__zs]

        # get the arrows components
        r1 = self.s_coordinate_system[0]
        r2 = self.s_coordinate_system[1]
        r3 = self.s_coordinate_system[2]
        r4 = self.s_coordinate_system[3]
        r5 = self.s_coordinate_system[4]
        r6 = self.s_coordinate_system[5]

        # adjust the position
        r1.setPos([self.origin[0], self.origin[1]])
        r2.setPos([r1.pos()[0] - r1.size()[0], r1.pos()[1]])
        r3.setPos([r1.pos()[0] - r1.size()[0], r1.pos()[1]])
        r4.setPos([self.origin[0], self.origin[1]])
        r5.setPos([r4.pos()[0], r4.pos()[1] - r4.size()[0]])
        r6.setPos([r4.pos()[0], r4.pos()[1] - r4.size()[0]])

        # adjust the size of the coordinate systems according to the view range of viewbox
        range = self.ui.plotWidget.plotItem.getViewBox().viewRange()[0]  # y and x has the same range
        range = np.abs(range[1] - range[0])

        r1.setSize([range / 10, 0], update=False)
        r2.setSize([range / 30, 0], update=False)
        r2.rotate(25, finish=False)
        r3.setSize([range / 30, 0], update=False)
        r3.rotate(-25, finish=False)

        r4.setSize([range / 10, 0], update=False)
        r5.setSize([range / 30, 0], update=False)
        r5.rotate(90 - 25, finish=False)
        r6.setSize([range / 30, 0], update=False)
        r6.rotate(90 + 25, finish=False)

        # adjust rotation
        if self.rotate:
            # get rotation matrix
            r = self._getR(self.__alphaR)

            # first arrow

            pos[0] -= self.origin[0]
            pos[1] -= self.origin[1]
            pos = r.dot(pos)
            r2.setPos(pos + self.origin, update=False)

            pos = [r1.pos()[0] - r1.size()[0], r1.pos()[1]]
            pos[0] -= self.origin[0]
            pos[1] -= self.origin[1]
            pos = r.dot(pos)
            r3.setPos(pos + self.origin, update=False)

            r1.setAngle(self.__alphaD + 180, update=False)
            r2.setAngle(self.__alphaD + 25, update=False)
            r3.setAngle(self.__alphaD - 25, update=False)

            # second arrow

            # pos = r4.pos()
            # pos -= self.origin
            # pos = r.dot(pos)
            # r4.setPos(pos + self.origin, update=False)

            pos = [r4.pos()[0], r4.pos()[1] - r4.size()[0]]

            pos[1] -= self.origin[1]

[docs]    def calcMiddlePoints(self):
        """
        calculates the coordinates of z to lower and upper cords and display them in the input fields
        """

        if self.calculation == "thin":
            self.__z0 = -self.__hs / 2
            self.__zu = self.__hs / 2
        else:
            self.__z0 = -self.__hs / 2 - self.__t0 / 2
            self.__zu = self.__hs / 2 + self.__tu / 2

        self.ui.lineEditzu.setText(str(round(self.__zu, 3)))

    def _getR(self, alpha):
        """
        gets the rotation matrix for 2D
        :param alpha: rotation angle in radian
        :return: 2x2 rotation matrix
        """

        # 2D rotation matrix
        r = np.array([[np.cos(alpha), -np.sin(alpha)], [np.sin(alpha), np.cos(alpha)]])
        return r
