[docs]def get_rmat(M, N):
    """
    Gets the 3D Rotation matrix of a plane defined with vector normal M.
    N will be the vector normal to the plane you want to rotate into.

    :param M: 3 component list
    :type M: double
    :param N: 3 component list
    :type N: double
    :return: 3x3 matrix
    :rtype: double
    """
    # Let M be the vector normal to your current plane, and N be the vector normal to the plane you want to rotate int
    M = np.array(M)
    N = np.array(N)
    # If M == N you can stop now and leave the original points unchanged
    if np.array_equal(np.absolute(M), np.absolute(N)):
        return None
    # Calculate the rotation angle
    costheta = np.dot(M, N) / (norm(M) * norm(N))
    # Calculate the rotation axis
    axis = np.cross(M, N) / norm(np.cross(M, N))
    c = costheta
    s = np.sqrt(1 - c * c)
    C = 1 - c
    x = axis[0]
    y = axis[1]
    z = axis[2]
    return np.array([[x * x * C + c, x * y * C - z * s, x * z * C + y * s],
                     [y * x * C + z * s, y * y * C + c, y * z * C - x * s],
                     [z * x * C - y * s, z * y * C + x * s, z * z * C + c]])




[docs]def transform(T, points):
    """
    Transforms each point in points matrix using the transformation matrix T.
    points should have row wise point coordinates.

    :param T: 3x3 Transformation Matrix
    :type T: double[][]
    :param points: nx3 points matrix
    :type points: double[][]
    :return: nx3 points matrix
    :rtype: double
    """
    out = np.zeros_like(points, dtype=np.double)
    for i in range(len(points)):
        out[i] = T.dot(points[i])
    return out



def fill_edof(edof):
    rows, columns = np.where(edof == 0)
    max_dof = np.amax(edof)
    for x, y in zip(rows, columns):
        edof[x, y] = max_dof + 1
        max_dof += 1
    return edof



[docs]def get_local_orientation(orientation):
    """
    Calculates all local axes orientation vectors according to v vector and
     normal vectors n = (node2-node1)/abs(node2-node1)
    Z axes is the projection of v on the plane with orthogonal vector X axes

    :return: x,y,z directions in each row
    """
    # get all n and v vectors
    # orientation = self.data_set.get_link_orientations()
    x = np.zeros((len(orientation), 3))  # x axes vectors
    y = np.zeros((len(orientation), 3))  # y axes vectors
    z = np.zeros((len(orientation), 3))  # z axes vectors
    index = 0
    for row in orientation:
        x_el = np.array([row[0], row[1], row[2]])
        v = np.array([row[3], row[4], row[5]])  # direction vector
        # normalize
        x_el = normalize(x_el.reshape((1, -1)))[0]
        v = normalize(v.reshape((1, -1)))[0]
        z_el = np.array(_project_onto_plane(v, x_el))
        z_el = normalize(z_el.reshape((1, -1)))[0]
        y_el = -np.cross(x_el, z_el)  # negative because left hand coordinate systems are being used
        x[index] = x_el
        y[index] = y_el
        z[index] = z_el
        index += 1
    return x, y, z




[docs]def assemble_global_f(loads, Dof, ndof):
    """Assemble global loads vector from nodes loads.
    Loads should contain nodes coordinates and the forces applied to each node [x,y,z, u,v,w, m_x,m_y,m_z]

    :param loads: Nodes loads array
    :type loads: float matrix
    :param Dof: nodes degrees of freedom
    :type Dof: integer matrix
    :param ndof: number of dofs n (defines the dimension of f)
    :type ndof: integer
    :return: global force vector (n x 1 vector)
    :rtype: float vector
    """
    # global f size is the last dof number
    f = np.zeros([ndof, 1])
    for load in loads:

        n_dofs = Dof[int(load[0]) - 1, :] - 1  # get dof index in global f for each node

        column_id = 0
        for dof_id in n_dofs:
            # exclude forces applied to nodes that are not connected to the structure
            if dof_id >= f.size:
                continue
            f[dof_id] += load[column_id + 1]  # superposition of each load vector component
            column_id += 1
    return f




[docs]def assemble_element_loads(loads, Edof):
    """Assemble element local loads vector from element loads.
    Loads should contain [link_id, qx, qy, qz, qw]

    :param loads: element loads
    :type loads: float matrix
    :param Edof: Element degree of freedom (Topology matrix)
    :type Edof: integer matrix
    :return: local forces
    :rtype: float matrix
    """
    eq = np.zeros([len(Edof), 4])
    for load in loads:
        eq[int(load[0]) - 1] = load[1:]
    return eq




[docs]def assemble_bc(nodes_bc, Dof):
    """Assemble boundary condition information in a vector.
    nodes_bc should contain [u_x,u_y,u_z,phi_x,phi_y,phi_z] for each node, which can only contain
    0 (free) or 1 (locked).
    The assembled bc vector could be ex. [1,2,5,6] that locks movement in those numbered directions.

    :param nodes_bc: boundary condition information from database
    :type nodes_bc: integer matrix
    :param Dof: nodes degrees of freedom
    :type Dof: integer matrix
    :return: bc vector
    :rtype: integer 1 x n vector
    """
    bc = []
    row_id = 0
    for bcond in nodes_bc:
        indices = np.where(bcond == 1)[0]
        for i in indices:
            bc.append(Dof[row_id, i])
        row_id += 1
    return np.array(bc)




[docs]def assemble_nodes_results(o_edof, nodes, a, Q):
    """Assemble nodes displacements and support forces according without joints dof results

    :param o_edof: origin edof containing zeros for free dof
    :type o_edof: integer matrix
    :param nodes: nodes coordinates
    :type nodes: float matrix
    :param a: nodes displacements
    :type a: numpy float vector
    :param Q: nodes support forces
    :type Q: numpy float vector
    :return: table with columns ['Node Nr.', 'ax', 'ay', 'az', 'aphi_x', 'aphi_y', 'aphi_z', 'x', 'y', 'z', 'Qx',
    'Qy','Qz', 'Qphi_x', 'Qphi_y', 'Qphi_z']
    :rtype: pandas Dataframe
    """
    # remove joint dof results
    n_joints = np.count_nonzero(o_edof == 0)
    if n_joints != 0:
        a = a[:-np.count_nonzero(o_edof == 0), 0]
        Q = Q[:-np.count_nonzero(o_edof == 0), 0]
    # reshape for 6 dof for each node
    a = np.reshape(a, (-1, 6))  # displacements
    Q = Q.reshape((-1, 6))  # reaction forces
    # results matrix
    results = np.c_[np.arange(1, len(nodes) + 1), a, nodes, Q]
    # values will be sorted for Node Nr. in case of existing joint nodes that has the same number as their origin nodes
    results = pd.DataFrame(results,
                           columns=['Node Nr.', 'ax', 'ay', 'az', 'aphi_x', 'aphi_y', 'aphi_z', 'x', 'y', 'z', 'Qx',
                                    'Qy',
                                    'Qz', 'Qphi_x', 'Qphi_y', 'Qphi_z'])
    results['Node Nr.'] = pd.to_numeric(results['Node Nr.'], downcast='integer')
    return results



def _project_onto_plane(v, n):
    """gets the projection of v onto a plane defined by n as its orthogonal vector.

    :param v: numpy vector in R3
    :param n: numpy normal vector in R3
    :return: numpy array
    """
    d = np.dot(v, n) / np.linalg.norm(n)
    p = [d * n[i] for i in range(len(n))]
    return [v[i] - p[i] for i in range(len(v))]
