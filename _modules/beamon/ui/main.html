[docs]    def append_to_log(self, msg):
        """Appends a given message to log and timestampes it with current datetime

        :param msg: string
        """
        if len(msg) == 0:
            return
        dt = datetime.now().strftime("%d-%m-%Y_%H-%M-%S")
        self.log.append(dt + '\t: ' + msg)
        self.display_message(msg)

[docs]    def open_database(self, path: str, overwrite: bool = False):
        """Opens a database in path. Opens a new empty Database if overwrite is true. if File do not exist and overwrite
        is false an exception will be thrown.

        :param path: path to database file
        :type path: string
        :param overwrite: True/False
        :type overwrite: boolean
        """
        if overwrite:
            # remove database first
            if os.path.isfile(path):
                os.remove(path)
            # open new database
            self.database = Database(ram=self.ram, path_to_save_database=path)
        else:
            # open existing database
            self.database = Database(path_to_database=path, ram=self.ram)

            # self.database.sort_node_indexes()
        # self.setWindowTitle(self.database.name)


[docs]    def open_database_diag(self):
        """Opens a user interface to load a database file
        """
        dlg = QFileDialog()
        path = os.getcwd()
        filter = "Database (*.db)"
        title = "open db file"
        fname = QFileDialog.getOpenFileName(dlg, title, path, filter)
        # init the database
        if fname[0] == "":
            # no file was chosen, init default empty database
            return
        else:
            try:
                self.database.close()
                self.open_database(path=fname[0])
                # self.database.sort_node_indexes()
                # close all visualizations if open
                for item in self.ui.mdiArea.subWindowList():
                    if item.windowTitle() == "Visualizer":
                        self.ui.mdiArea.setActiveSubWindow(item)
                        self.ui.mdiArea.closeActiveSubWindow()
                        self.startVisualizer()
                self.append_to_log("project " + str(path) + " loaded.")
            except Exception as e:
                self.append_to_log(str(e))

        """Saves the database to a database file
        """
        dlg = QFileDialog()
        path = os.getcwd()
        filter = "Database (*.db)"
        title = "Save project as"
        fname = dlg.getSaveFileName(dlg, title, path, filter)
        if fname[0] == "":
            # no file was chosen, init default empty database
            pass
        else:
            if fname[0] == self.database.name and not self.database.ram:
                # do nothing
                pass
            else:
                if os.path.isfile(fname[0]):
                    os.remove(fname[0])
                self.database.backup(fname[0])

[docs]    def closeEvent(self, event):
        """
        controls closing signal
        :param event: the closing event
        """
        if self.confirm_exit:
            msg = QMessageBox()
            msg.setIcon(QMessageBox.Warning)
            icon = QIcon()
            icon.addPixmap(QPixmap(":/tree/cancel.png"), QIcon.Normal, QIcon.Off)
            msg.setWindowIcon(icon)
            msg.setText("Are you sure you want to close?")
            msg.setWindowTitle("Closing app")
            msg.setStandardButtons(QMessageBox.Ok | QMessageBox.Cancel)
            retval = msg.exec_()
            if retval == QMessageBox.Cancel:
                event.ignore()
            else:

                if self.project_browser is not None:
                    self.project_browser.close()
                if self.vis_manual is not None:
                    self.vis_manual.close()
                event.accept()
        else:
            self.notification.close()
            if self.project_browser is not None:
                self.project_browser.close()
            if self.vis_manual is not None:
                self.vis_manual.close()
            event.accept()

[docs]    def startVisualizer(self, m_id: int = None):
        """starts the application for visualising models.

        :param m_id: model number
        :type m_id: integer
        """
        model_name = self.database.get_model(m_id)[0][1]
        # singleton models (call only once for every model)
        visualizer_widget = self.visualizers.get(m_id)
        if visualizer_widget is not None:  # key already used
            if visualizer_widget.isVisible():
                # activate sub window
                for item in self.ui.mdiArea.subWindowList():
                    if item.windowTitle() == model_name:
                        self.ui.mdiArea.setActiveSubWindow(item)
                        return

        # create the visualizer
        visualizer = VisualizerMainForm(self.ui, self, m_id)
        icon = QIcon()
        icon.addPixmap(QPixmap(":/tree/loads.png"), QIcon.Normal, QIcon.Off)
        visualizer.setWindowIcon(icon)
        visualizer.setWindowTitle(model_name)
        # add to hash map
        self.visualizers[m_id] = visualizer
        # make it a subwindow
        sub_window = QMdiSubWindow()
        sub_window.setWidget(visualizer)
        self.ui.mdiArea.addSubWindow(sub_window)
        visualizer.show()
        self.create_window_action(m_id, model_name)

        # self.ui.toolBar.addAction(self.visualizerAction)
        # to maximize the window on start
        # self.visualizer.setWindowState(Qt.WindowMaximized)
