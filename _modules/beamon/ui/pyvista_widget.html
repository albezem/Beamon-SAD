[docs]    def update_global(self, m_id: int, reset_view=True):
        """Central update for all visualization objects based on their hierarchic dependencies and their data sets.
        @see machine state diagram 'user interaction' in documentation

        :param m_id: model number
        :type m_id: integer
        :param reset_view: reset camera view
        :type reset_view: boolean
        """

        self.update_grid()
        # nodes
        self.update_nodes(m_id)
        # bc
        self.update_boundary_conditions(m_id)
        # loads
        self.update_loads(m_id)
        # lineloads
        self.update_lineloads(m_id)
        # elements
        self.update_elements(m_id)
        self.update_elements_orientation(m_id)
        # joints
        self.update_joints(m_id)
        # labels
        self.update_labels(reset_view=False, m_id=m_id)
        # results
        self.update_results(m_id)

        # camera view
        if reset_view:
            self.reset_view()


    def reset_view(self):
        if self.dim == 2:
            self.enable_image_style()
            self.enable_parallel_projection()
            self._key_2_pressed()
        else:
            self.enable_trackball_style()
            self.disable_parallel_projection()
            # change focal point of the camera
            self.camera.SetFocalPoint(self.model_center)


[docs]    def update_grid(self):
        """
        Update grid data according to grid settings such as vector normal, size, tick and type
        """
        # init a 2d Grid for drawing
        size = self._grid_size

        # make 1. quadrant grid starting from origin
        XX1, ZZ1 = np.meshgrid(np.arange(self.grid_origin[0], size, self.grid_xtick),
                               np.arange(self.grid_origin[2], size, self.grid_ytick))
        XX1 = XX1.reshape((XX1.shape[0] * XX1.shape[1], 1))
        ZZ1 = ZZ1.reshape((ZZ1.shape[0] * ZZ1.shape[1], 1))
        YY1 = np.zeros_like(ZZ1)
        points = np.concatenate((XX1, YY1, ZZ1), axis=1)
        rmat = get_rmat([0, 1, 0], self.grid_n)
        if not rmat is None:
            # rotate points using rotation matrix
            points = transform(rmat, points)
        # self.update_coordinates(points=points, mesh=self.grid_actor)
        self.remove_actor(self.grid_actor)
        self.grid_points = pv.PolyData(points)
        self.grid_actor = self.add_mesh(self.grid_points, reset_camera=False, opacity=0.5, name="gird",
                                        render_points_as_spheres=True, point_size=3)

[docs]    def update_boundary_conditions(self, m_id):
        """scatter two blue cones for momentum and orange blue cones for translation on each node with bc
        """
        self.remove_actor('bc_translation')
        self.remove_actor('bc_momentum1')
        self.remove_actor('bc_momentum2')
        # get bc information from all nodes (including coordinates
        dof = self.dataset.get_all_dof(m_id)
        if dof.size == 0 or not self.show_bc:
            return
        v1 = np.array([1, 0, 0])  # x direction
        v2 = np.array([0, 1, 0])  # y direction
        v3 = np.array([0, 0, 1])  # z direction
        # translation
        height1 = 0.2
        centers_trans = np.zeros((0, 3))
        directions_trans = np.zeros((0, 3))
        # momentum1
        height21 = 0.4
        centers_mom1 = np.zeros((0, 3))
        directions_mom1 = np.zeros((0, 3))
        # momentum2
        height22 = 0.5
        centers_mom2 = np.zeros((0, 3))
        directions_mom2 = np.zeros((0, 3))
        for bc in dof:
            # x direction
            if bc[3] == 1:
                # append x direction translation bc
                centers_trans = np.vstack((centers_trans, bc[0:3] - (height1 / 2) * v1))
                directions_trans = np.vstack((directions_trans, v1))
            if bc[6] == 1:
                # append x direction momentum bc (two cones)
                centers_mom1 = np.vstack((centers_mom1, bc[0:3] - (height21 / 2) * v1))
                directions_mom1 = np.vstack((directions_mom1, v1))
                centers_mom2 = np.vstack((centers_mom2, bc[0:3] - (height22 / 2) * v1))
                directions_mom2 = np.vstack((directions_mom2, v1))
            # y direction
            if bc[4] == 1:
                # append x direction translation bc
                centers_trans = np.vstack((centers_trans, bc[0:3] - (height1 / 2) * v2))
                directions_trans = np.vstack((directions_trans, v2))
            if bc[7] == 1:
                # append x direction momentum bc (two cones)
                centers_mom1 = np.vstack((centers_mom1, bc[0:3] - (height21 / 2) * v2))
                directions_mom1 = np.vstack((directions_mom1, v2))
                centers_mom2 = np.vstack((centers_mom2, bc[0:3] - (height22 / 2) * v2))
                directions_mom2 = np.vstack((directions_mom2, v2))
            # z direction
            if bc[5] == 1:
                # append x direction translation bc
                centers_trans = np.vstack((centers_trans, bc[0:3] - (height1 / 2) * v3))
                directions_trans = np.vstack((directions_trans, v3))
            if bc[8] == 1:
                # append x direction momentum bc (two cones)
                centers_mom1 = np.vstack((centers_mom1, bc[0:3] - (height21 / 2) * v3))
                directions_mom1 = np.vstack((directions_mom1, v3))
                centers_mom2 = np.vstack((centers_mom2, bc[0:3] - (height22 / 2) * v3))
                directions_mom2 = np.vstack((directions_mom2, v3))
        # translation
        if centers_trans.size != 0:
            data = pv.PolyData(centers_trans)
            data['vectors'] = directions_trans
            self.add_mesh(
                data.glyph(geom=pv.Cone(resolution=10, height=height1, radius=0.1, capping=False), orient='vectors'),
                color="#ff751a", name='bc_translation', reset_camera=False, show_edges=True)
        if centers_mom1.size != 0 or centers_mom2.size != 0:
            # momentum1
            data = pv.PolyData(centers_mom1)
            data['vectors'] = directions_mom1
            self.add_mesh(
                data.glyph(geom=pv.Cone(resolution=10, height=height21, radius=0.1, capping=False), orient='vectors'),
                color="#0000ff", name='bc_momentum1', reset_camera=False, show_edges=True)
            # momentum2
            data = pv.PolyData(centers_mom2)
            data['vectors'] = directions_mom2
            self.add_mesh(
                data.glyph(geom=pv.Cone(resolution=10, height=height22, radius=0.1, capping=False), orient='vectors'),
                color="#0000ff", name='bc_momentum2', reset_camera=False, show_edges=True)


    def update_loads(self, m_id: int):
        self.remove_actor("forces")
        self.remove_actor("momentum")
        if not self.show_node_loads:
            return
        # get loads data
        loads = self.dataset.get_loads_for_plotting(m_id)  # (x,y,z,u,v,w, m_x,m_y,m_z)
        if loads.size == 0:
            return
        # norm vectors
        loads[:, 3:6] = normalize(loads[:, 3:6])
        # draw forces
        cent = loads[:, :3] - loads[:, 3: 6] * self.force_mag

        self.add_arrows(cent=cent, direction=loads[:, 3:6], mag=self.force_mag, color="#009900", name="forces",
                        reset_camera=False)
        # draw momentum
        cent = np.empty((0, 3))
        direction = np.empty((0, 3))

        # get momentum in x direction
        m_x = loads[:, 6][np.nonzero(loads[:, 6])]  # get momentum that are none zero
        if not m_x.size == 0:
            vectorx = np.zeros(shape=(len(m_x), 3))
            vectorx[:, 0] = 1
            new_cent = loads[:, :3][np.nonzero(loads[:, 6])]  # first arrow cents
            new_cent = np.concatenate((new_cent, new_cent + 0.2 * vectorx))  # second arrow cents just translated
            vectorx = np.concatenate((vectorx, vectorx))
            # add to global cent and direction
            cent = np.concatenate((cent, new_cent))
            direction = vectorx
        # get momentum in y direction
        m_y = loads[:, 7][np.nonzero(loads[:, 7])]  # get momentum that are none zero
        if not m_y.size == 0:
            vectorx = np.zeros(shape=(len(m_y), 3))
            vectorx[:, 1] = 1
            new_cent = loads[:, :3][np.nonzero(loads[:, 7])]  # first arrow cents
            new_cent = np.concatenate((new_cent, new_cent + 0.2 * vectorx))  # second arrow cents just translated
            vectorx = np.concatenate((vectorx, vectorx))
            # add to global cent and direction
            cent = np.concatenate((cent, new_cent))
            direction = np.concatenate((direction, vectorx))
        # get momentum in z direction
        m_z = loads[:, 8][np.nonzero(loads[:, 8])]  # get momentum that are none zero
        if not m_z.size == 0:
            vectorx = np.zeros(shape=(len(m_z), 3))
            vectorx[:, 2] = 1
            new_cent = loads[:, :3][np.nonzero(loads[:, 8])]  # first arrow cents
            new_cent = np.concatenate((new_cent, new_cent + 0.2 * vectorx))  # second arrow cents just translated
            vectorx = np.concatenate((vectorx, vectorx))
            # add to global cent and direction
            cent = np.concatenate((cent, new_cent))
            direction = np.concatenate((direction, vectorx))

        if not cent.size == 0:
            self.add_arrows(cent=cent, direction=direction, mag=self.momentum_mag, color="#a2ab00", name="momentum",
                            reset_camera=False)

    def update_lineloads(self, m_id: int):
        self.remove_actor("lineload_line")
        self.remove_actor("lineload_x")
        self.remove_actor("lineload_y")
        self.remove_actor("line_lineload_y")
        self.remove_actor("lineload_z")
        self.remove_actor("line_lineload_z")
        self.remove_actor("lineload_w")
        if not self.show_line_loads:
            return
        # number of evaluation points
        n = 20
        # scaling
        mag_x = self.force_mag - 0.4
        mag_y = mag_x
        mag_z = mag_y
        mag_w = self.force_mag - 0.3
        # get lineloads data
        data = self.dataset.get_lineloads_for_plotting(m_id)
        if data.size == 0:
            return
        # for saving center coordinates with directions
        direction_x = np.zeros((0, 6))
        direction_y = np.zeros((0, 6))
        direction_z = np.zeros((0, 6))
        direction_w = np.zeros((0, 6))
        # for saving line segments coordinates
        lines_y = np.zeros((0, 3))
        lines_z = np.zeros((0, 3))
        e_id = 1
        for load in data:
            # get evaluation points for each element
            p1 = load[1:4]  # starting node coordinates
            p2 = load[4:7]  # ending node coordinates
            length = np.linalg.norm(p2 - p1)  # element length
            ox, oy, oz = self.dataset.get_link_orientation(load[0])  # element orientation
            xi = np.linspace(0, length, n)  # evaluation points along the local x axis
            n_cent = np.c_[  # coordinates along the element
                np.full((len(xi),), p1[0]) + xi * ox[0][0],
                np.full((len(xi),), p1[1]) + xi * ox[0][1],
                np.full((len(xi),), p1[2]) + xi * ox[0][2]
            ]
            if load[7] != 0:
                direction_x = np.vstack(
                    (direction_x, np.c_[np.full((len(xi), 3), np.sign(load[7]) * ox), n_cent]))
            if load[8] != 0:
                direction_y = np.vstack(
                    (direction_y,
                     np.c_[np.full((len(xi), 3), np.sign(load[8]) * oy), n_cent - np.sign(load[8]) * mag_y * oy[0]]))
                # add line segment coordinates
                lines_y = np.vstack(
                    (lines_y, p1 - np.sign(load[8]) * mag_y * oy[0],
                     p2 - np.sign(load[8]) * mag_y * oy[0]))
            if load[9] != 0:
                direction_z = np.vstack(
                    (direction_z,
                     np.c_[np.full((len(xi), 3), np.sign(load[9]) * oz), n_cent - np.sign(load[9]) * mag_z * oz[0]]))
                # add line segment coordinates
                lines_z = np.vstack(
                    (lines_z, p1 - np.sign(load[9]) * mag_z * oz[0],
                     p2 - np.sign(load[9]) * mag_z * oz[0]))
            if load[10] != 0:
                xi = np.linspace(0, length, int(n / 2))
                n_cent = np.c_[
                    np.full((len(xi),), p1[0]) + xi * ox[0][0],
                    np.full((len(xi),), p1[1]) + xi * ox[0][1],
                    np.full((len(xi),), p1[2]) + xi * ox[0][2]
                ]
                n_cent2 = np.c_[
                    np.full((len(xi),), p1[0]) + (xi + 0.1) * ox[0][0],
                    np.full((len(xi),), p1[1]) + (xi + 0.1) * ox[0][1],
                    np.full((len(xi),), p1[2]) + (xi + 0.1) * ox[0][2]
                ]
                direction_w = np.vstack((direction_w,
                                         np.c_[np.full((len(xi), 3), np.sign(load[10]) * ox), n_cent],
                                         np.c_[np.full((len(xi), 3), np.sign(load[10]) * ox), n_cent2]))

            e_id += 1
        if direction_x.size != 0:
            self.add_arrows(cent=direction_x[:, 3:], direction=direction_x[:, 0:3], mag=mag_x,
                            color="#009900", name="lineload_x", reset_camera=False)
        if direction_y.size != 0:
            self.add_arrows(cent=direction_y[:, 3:], direction=direction_y[:, 0:3], mag=mag_y,
                            color="#009900", name="lineload_y", reset_camera=False)
            self.add_mesh(pv.line_segments_from_points(lines_y),
                          reset_camera=False, name="line_lineload_y", line_width=2, color='#009900')
        if direction_z.size != 0:
            self.add_arrows(cent=direction_z[:, 3:], direction=direction_z[:, 0:3], mag=mag_z,
                            color="#009900", name="lineload_z", reset_camera=False)
            self.add_mesh(pv.line_segments_from_points(lines_z),
                          reset_camera=False, name="line_lineload_z", line_width=2, color='#009900')
        if direction_w.size != 0:
            self.add_arrows(cent=direction_w[:, 3:], direction=direction_w[:, 0:3], mag=mag_w,
                            color="#a2ab00", name="lineload_w", reset_camera=False)

[docs]    def update_support_forces(self, m_id: int):
        """Updates nodes support forces
        """
        self.remove_actor("nodes_sup_forces_trans")  # translation
        self.remove_actor("nodes_sup_forces_mom")  # momentum
        if (self.results_view_map['nodes sup forces trans'] == 0 and self.results_view_map[
            'nodes sup forces mom'] == 0) or self.m_id not in self.dataset.nodes_results.keys():
            return
        # get results from nodes with bc
        bc = self.dataset.get_nodes_indexes_with_bc(m_id)
        if False in self.dataset.nodes_results[self.m_id]['Node Nr.'].isin(
                bc):  # check if bc has been altered after simulation
            return
        data = self.dataset.nodes_results[self.m_id].set_index('Node Nr.').loc[bc]
        if data.isnull().values.any():
            data.fillna(0, inplace=True)
        # get nodes coordinates as centers
        cent_data = data.iloc[:, 6:9].to_numpy(dtype=float)
        cent = np.vstack((cent_data, cent_data, cent_data))

        # translation
        direction1 = data.iloc[:, 9:12].to_numpy(dtype=float)
        zeros_vec = np.zeros((len(direction1), 1))
        direction1 = np.vstack((np.c_[direction1[:, 0], zeros_vec, zeros_vec],
                                np.c_[zeros_vec, direction1[:, 1], zeros_vec],
                                np.c_[zeros_vec, zeros_vec, direction1[:, 2]]))
        # momentum
        direction2_data = data.iloc[:, 12:].to_numpy(dtype=float)
        zeros_vec = np.zeros((len(direction2_data), 1))
        direction2 = np.vstack((
            np.c_[direction2_data[:, 0], zeros_vec, zeros_vec],
            np.c_[direction2_data[:, 0], zeros_vec, zeros_vec],
            np.c_[zeros_vec, direction2_data[:, 1], zeros_vec],
            np.c_[zeros_vec, direction2_data[:, 1], zeros_vec],
            np.c_[zeros_vec, zeros_vec, direction2_data[:, 2]],
            np.c_[zeros_vec, zeros_vec, direction2_data[:, 2]]
        ))
        cent2 = np.vstack((
            cent_data, cent_data + np.c_[0.2 * np.sign(direction2_data[:, 0]), zeros_vec, zeros_vec],
            cent_data, cent_data + np.c_[zeros_vec, 0.2 * np.sign(direction2_data[:, 1]), zeros_vec],
            cent_data, cent_data + np.c_[zeros_vec, zeros_vec, 0.2 * np.sign(direction2_data[:, 2])]
        ))

        # translation
        # normalize and scale
        norms = np.array([np.linalg.norm(v) for v in direction1])
        scales = norms / np.amax(norms)
        direction1 = np.array([v * s for v, s in zip(normalize(direction1), scales)])
        if self.results_view_map['nodes sup forces trans'] != 0:
            self.add_arrows(cent=cent, direction=direction1, color="#ff751a",
                            name="nodes_sup_forces_trans", reset_camera=False)
        # momentum
        norms = np.array([np.linalg.norm(v) for v in direction2])
        scales = norms / np.amax(norms)
        direction2 = np.array([v * s for v, s in zip(normalize(direction2), scales)])
        if self.results_view_map['nodes sup forces mom'] != 0:
            self.add_arrows(cent=cent2, direction=direction2, color="#0000ff",
                            name="nodes_sup_forces_mom", reset_camera=False)

    def _get_segments_data(self, points, scales=None):
        """creates line segment points for each element evaluation points keeping segmentation information between
        elements.
        If points are [e1:p1,e1:p2,e1:p3,e2:p4,e2:p5,e2:p6]
        line segment points will be [p1,p2, p2,p3 ,p4,p5, p5,p6]

        :param points: [e id, x,y,z]
        :type points: float matrix
        :param scales: optional scale value for each evaluation point
        :type scales: float vector
        :return: segments points and scales if given
        :rtype: float array(s)
        """
        if not points.shape[1] != 3:
            raise ValueError("wrong input")
        # if no scales where given
        if scales is None:
            o_points = np.zeros((points.shape[0] * 2 - 2 * int(points[-1, 0]), points.shape[1] - 1))
            out_id = 0
            for in_id in range(0, len(points) - 1):
                e_id = points[in_id, 0]
                n_e_id = points[in_id + 1, 0]
                if e_id != n_e_id:
                    continue
                o_points[out_id] = points[in_id, 1:]
                o_points[out_id + 1] = points[in_id + 1, 1:]

                out_id += 2
            return o_points
        else:
            if not len(scales.shape) == 1 or not scales.shape[0] == points.shape[0]:
                raise ValueError("wrong input")

            if points.shape[0] != scales.shape[0]:
                raise Exception("Dimensions unmatched")
            o_points = np.zeros((points.shape[0] * 2 - 2 * int(points[-1, 0]), points.shape[1] - 1))
            o_scales = np.zeros((scales.shape[0] * 2 - 2 * int(points[-1, 0]), 1))
            out_id = 0
            for in_id in range(0, len(points) - 1):
                e_id = points[in_id, 0]
                n_e_id = points[in_id + 1, 0]
                if e_id != n_e_id:
                    continue
                o_points[out_id] = points[in_id, 1:]
                o_points[out_id + 1] = points[in_id + 1, 1:]

                o_scales[out_id] = scales[in_id]
                o_scales[out_id + 1] = scales[in_id + 1]

                out_id += 2

            return o_points, o_scales
