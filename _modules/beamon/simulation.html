[docs]    def simulate(self, m_id: int, n: int = 20):
        """Run the simulation according to data in the database.

        :param m_id: model number
        :type m_id: integer
        :param n: number of evaluation points
        :type n: integer
        """
        # nodes coordinates
        nodes_data = self.database.get_nodes(m_id)
        # get topology matrix
        origin_Edof = self.database.get_edof_for_simulation(m_id)
        if origin_Edof.size == 0:
            raise SimulationError("No topology defined (see if elements where defined)")
        # create nodes dof numbers matrix
        Dof = cal.createdofs(len(nodes_data), 6)
        # get boundary conditions
        bc = beamcalc.assemble_bc(self.database.get_bc(m_id), Dof)
        # fill edof matrix with new dof numbers
        Edof = beamcalc.fill_edof(origin_Edof.copy())

        # make joints. Creates new topology with cloned nodes (with new dof) for joints
        # Edof, nodes, Dof, bc, joints_mapping = beamcalc.make_joints(origin_Edof.copy(), nodes_data, origin_Dof, 6, bc)

        # extract element coordinates
        static_edof = self.database.get_static_edof(m_id)
        Ex, Ey, Ez = cal.coordxtr(static_edof, nodes_data, Dof)
        # dof count
        number_of_dofs = np.amax(origin_Edof) + np.count_nonzero(origin_Edof == 0)
        if any(x > number_of_dofs for x in bc):
            raise SimulationError("Boundary conditions not connected to structure")
        # preallocate stiffness matrix
        K = np.zeros([number_of_dofs, number_of_dofs])
        # get global loads vector from nodes loads
        f = beamcalc.assemble_global_f(self.database.get_loads_for_simulation(m_id).to_numpy(), Dof, number_of_dofs)
        # get element local loads vector from element loads
        eq = beamcalc.assemble_element_loads(self.database.get_all_lineloads_for_simulation(m_id).to_numpy(), Edof)
        # get elements properties
        ep = self.database.get_links_profiles(m_id)
        # raise error if some elements has no profiles defined
        if len(ep) != len(origin_Edof):
            raise SimulationError("Some elements have no profiles defined")
        # get elements orientations (z axis is important for calculation others for local to global transformation)
        eo_x, eo_y, eo_z = self.database.get_link_orientations(m_id)

        # Simulation part
        #  calculate element stiffness matrices and assemble them in global stiffness matrix
        for elx, ely, elz, eltopo, elo, elp, elq in zip(Ex, Ey, Ez, Edof[:, 2:], eo_z, ep, eq):
            Ke, fe = cal.beam3e(elx, ely, elz, elo, elp, elq.tolist())
            K, f = cal.assem(eltopo, K, Ke, f, fe)
        #  solve system of equations according to boundary conditions
        #  calculating nodes displacement and support forces
        try:
            a, r = cal.solveq(K, f, bc)
        except LinAlgError:
            raise SimulationError("Singular system of equations")
        # assembling them into a dataframe
        self.database.nodes_results[m_id] = beamcalc.assemble_nodes_results(origin_Edof, nodes_data, np.array(a), np.array(r))
        self.database.global_displaced_nodes[m_id] = pd.DataFrame(
            self.database.nodes_results[m_id].iloc[:, 7:10].to_numpy() + self.database.nodes_results[m_id].iloc[:, 1:4].to_numpy(),
            columns=['x', 'y', 'z'])
        # extract element displacements from nodes displacements without joints dof results
        ed = cal.extractEldisp(Edof[:, 2:], a)

        # for local element displacements and inner forces
        e_disp = np.zeros((0, 6))
        e_sec = np.zeros((0, 8))
        # for global elements evaluation points
        g_e_points = np.zeros((0, 4))  # [e_id, x, y, z]
        # for global displaced elements points
        g_e_disp = np.zeros((0, 4))  # [e_id, x, y, z]
        # for statistics for each element
        e_stats = np.zeros((0, 13))

        e_id = 1  # element number
        for elx, ely, elz, elo_x, elo_y, elo_z, elp, eld, elq in zip(Ex, Ey, Ez, eo_x, eo_y, eo_z, ep, ed, eq):
            # do calculation
            es, edi, eci = cal.beam3s(elx, ely, elz, elo_z, elp, eld, elq.tolist(), n)
            # save output
            e_disp = np.vstack((e_disp, np.c_[np.full(n, e_id), eci, edi]))
            e_sec = np.vstack((e_sec, np.c_[np.full(n, e_id), eci, es]))
            # extract new statistics information form results
            temp = np.array([e_id,
                             np.amin(es[:, 0]), np.amax(es[:, 0]),
                             np.amin(es[:, 1]), np.amax(es[:, 1]),
                             np.amin(es[:, 2]), np.amax(es[:, 2]),
                             np.amin(es[:, 3]), np.amax(es[:, 3]),
                             np.amin(es[:, 4]), np.amax(es[:, 4]),
                             np.amin(es[:, 5]), np.amax(es[:, 5]),
                             ])
            e_stats = np.vstack((e_stats, temp))

            # transform local points to global points:

            # calculate new global evaluation points (p1 + eci * elo_x)
            g_points = np.zeros((len(eci), 4))
            idx = 0
            for x in eci:
                g_points[idx] = np.hstack((e_id, np.array([elx[0], ely[0], elz[0]]) + x[0] * elo_x))
                idx += 1
            g_e_points = np.vstack((g_e_points, g_points))

            # transform local displacements to global displacements
            g_displacments = np.zeros((len(eci), 4))
            Tl = np.c_[elo_x, elo_y, elo_z]  # local transformation matrix
            T = np.matmul(self.Tg, Tl)  # total transformation matrix
            idx = 0
            for _ in eci:
                local_disp = edi[idx, :3]
                global_disp = T.dot(local_disp)
                # local_point = np.array([x[0], 0, 0]) + edi[id, :3]
                # global_point = T.dot(local_point) + np.array([elx[0], ely[0], elz[0]])
                # g_displacments[id] = np.hstack((e_id, global_point))
                g_displacments[idx] = np.hstack((e_id, global_disp))
                idx += 1
            g_e_disp = np.vstack((g_e_disp, g_displacments))

            e_id += 1
        self.database.local_element_displacements[m_id] = pd.DataFrame(e_disp,
                                                                 columns=['Element Nr.', 'xi', 'u', 'v', 'w', 'phi'])
        self.database.local_element_section_forces[m_id] = pd.DataFrame(e_sec,
                                                                  columns=['Element Nr.', 'xi', 'N', 'Qy', 'Qz', 'T',
                                                                           'My',
                                                                           'Mz'])
        self.database.element_stats[m_id] = pd.DataFrame(e_stats, columns=['Element Nr.',
                                                                     'Min N', 'Max N',
                                                                     'Min Vy', 'Max Vy',
                                                                     'Min Vz', 'Max Vz',
                                                                     'Min T', 'Max T',
                                                                     'Min My', 'Max My',
                                                                     'Min Mz', 'Max Mz'])
        self.database.global_element_evaluation_points[m_id] = pd.DataFrame(g_e_points,
                                                                      columns=['Element Nr.', 'x', 'y', 'z'])
        self.database.global_element_displacements[m_id] = pd.DataFrame(g_e_disp,
                                                                  columns=['Element Nr.', 'u', 'v', 'w'])
